pub async fn command_handler(bot: &mut UserBot, message: &mut gramme::types::Message) -> Result<(), UserBotError> {
    return match message.text().split_whitespace().next() {
        Some("!id") => plugins::id_handler(bot, message).await,
Some("!info") => plugins::info_handler(bot, message).await,
Some("!ping") => plugins::ping_handler(bot, message).await,
Some("!note") => plugins::add_note_handler(bot, message).await,
Some("!get") => plugins::get_note_handler(bot, message).await,
Some("!clear") => plugins::rm_notes_handler(bot, message).await,
Some("!notes") => plugins::all_notes_handler(bot, message).await,
Some("!ban") => plugins::ban_handler(bot, message).await,
Some("!unban") => plugins::unban_handler(bot, message).await,
Some("!kick") => plugins::kick_handler(bot, message).await,
Some("!mute") => plugins::mute_handler(bot, message).await,
Some("!unmute") => plugins::unmute_handler(bot, message).await,
Some("!getsticker") => plugins::get_sticker_handler(bot, message).await,
Some("!kang") => plugins::kang_handler(bot, message).await,
Some("!del") => plugins::msg_del(bot, message).await,
Some("!purge") => plugins::purge_msg(bot, message).await,
Some("!owo") => plugins::owo_handler(bot, message).await,
Some("!stretch") => plugins::stretch_handler(bot, message).await,
Some("!vapor") => plugins::vapor_handler(bot, message).await,
Some("!mock") => plugins::mock_handler(bot, message).await,
Some("!zalgo") => plugins::zalgo_handler(bot, message).await,
Some("!pp") => plugins::pp_handler(bot, message).await,
Some("!pin") => plugins::pin_handler(bot, message).await,
Some("!unpin") => plugins::unpin_handler(bot, message).await,
Some("!promote") => plugins::promote_handler(bot, message).await,
Some("!demote") => plugins::demote_handler(bot, message).await,
Some("!hello") => plugins::hello(bot, message).await,
        _ => Ok(()),
    }
}
